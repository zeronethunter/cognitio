set(SUBMODULE_NAME ${CMAKE_PROJECT_NAME}_block_storage)
set(${SUBMODULE_NAME}_SOURCES block_storage.cpp)

add_library(
        ${SUBMODULE_NAME} STATIC ${${SUBMODULE_NAME}_SOURCES}
)

target_link_libraries(
        ${SUBMODULE_NAME} PUBLIC
        ${CMAKE_PROJECT_NAME}_repo
        ${CMAKE_PROJECT_NAME}_multiformats
        ${CMAKE_PROJECT_NAME}_common
)

if (ENABLE_TESTING MATCHES ON)
    add_libs_for_testing(${SUBMODULE_NAME})
endif ()

include_directories(
        PUBLIC $<INSTALL_INTERFACE:include>
        $<BUILD_INTERFACE:${CMAKE_SOURCE_DIR}/include>
)
